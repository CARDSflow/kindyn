195 def getPedalPositions(numSamples):
196  # Format: [[pedal_angle_1, x, y, z], [pedal_angle_2, x, y, z], ...]
197  capturedPositions = []
198  angularStep = 2.0 * math.pi / numSamples
199  for sampleIterator in range(numSamples + 1):
200  thisPedalAngle = sampleIterator * angularStep
201  thisXVal = PEDAL_CENTER_OFFSET_X + math.cos(thisPedalAngle) * PEDAL_RADIUS
202  thisYVal = PEDAL_CENTER_OFFSET_Y
203  thisZVal = PEDAL_CENTER_OFFSET_Z + math.sin(thisPedalAngle) * PEDAL_RADIUS
204  capturedPositions.append([thisPedalAngle, thisXVal, thisYVal, thisZVal])
205 
206  return capturedPositions
