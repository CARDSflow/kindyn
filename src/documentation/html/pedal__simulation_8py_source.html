523 def evaluate_current_pedal_angle(current_point):
524  current_x = current_point[ 0 ] - PEDAL_CENTER_OFFSET_X
525  current_y = current_point[ 1 ] - PEDAL_CENTER_OFFSET_Y
526 
527  if current_x > 0 and current_y > 0:
528  return np.arctan(current_y / current_x)
529  elif current_x < 0 < current_y:
530  return np.arctan(current_y / current_x) + np.pi
531  elif current_x < 0 and current_y < 0:
532  return np.arctan(current_y / current_x) + np.pi
533  elif current_x > 0 > current_y:
534  return np.arctan(current_y / current_x) + 2 * np.pi
535 
536  elif current_x == 0 and current_y > 0:
537  return np.pi / 2
538  elif current_x == 0 and current_y < 0:
539  return np.pi * 3 / 2
540  elif current_x > 0 and current_y == 0:
541  return 0
542  elif current_x < 0 and current_y == 0:
543  return np.pi
