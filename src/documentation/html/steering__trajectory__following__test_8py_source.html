221 def computeSteeringAngles():
222 
223  global _steeringAngles
224  global NUM_STEERING_ANGLES
225 
226  if NUM_STEERING_ANGLES % 2 == 0:
227  print("ERROR: NUM_STEERING_ANGLES must be odd number")
228  return
229 
230  numSymmetricSteeringAngles = (NUM_STEERING_ANGLES - 1) / 2
231 
232  for sizeIterator in range(NUM_STEERING_ANGLES):
233  _steeringAngles.append(0.0)
234 
235  _steeringAngles[int(NUM_STEERING_ANGLES / 2)] = 0.0 # Redundant
236  for steeringAngleIterator in range(1, numSymmetricSteeringAngles + 1, 1):
237  thisAngle = steeringAngleIterator * (MAX_TURNING_ANGLE / numSymmetricSteeringAngles)
238  _steeringAngles[numSymmetricSteeringAngles + steeringAngleIterator] = thisAngle
239  _steeringAngles[numSymmetricSteeringAngles - steeringAngleIterator] = (-1) * thisAngle
