222  global _steeringAngles
223  global NUM_STEERING_ANGLES
224 
225  if NUM_STEERING_ANGLES % 2 == 0:
226  print("ERROR: NUM_STEERING_ANGLES must be odd number")

228 
229  numSymmetricSteeringAngles = (NUM_STEERING_ANGLES - 1) / 2

232  _steeringAngles.append(0.0)

234  _steeringAngles[int(NUM_STEERING_ANGLES / 2)] = 0.0 # Redundant
235  for steeringAngleIterator in range(1, numSymmetricSteeringAngles + 1, 1):
236  thisAngle = steeringAngleIterator * (MAX_TURNING_ANGLE / numSymmetricSteeringAngles)

238  _steeringAngles[numSymmetricSteeringAngles - steeringAngleIterator] = (-1) * thisAngle
