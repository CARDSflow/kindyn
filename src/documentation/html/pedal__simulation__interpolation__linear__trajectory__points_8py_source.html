402 def computeVelocitySetpoint(jointName, endPos, startTime, currTime, endTime):
403 
404  global _jointsControlData
405  global PRINT_DEBUG
406  global JOINT_TRAJECTORY_ERROR_TOLERANCE
407  global JOINT_VELOCITY_FACTOR
408 
409  currPos = getJointPosition(jointName)

411  jointError = goalPos - currPos
412 
413  if _jointsControlData[jointName]["bool_update_iv"]:
414  if currTime < endTime:
415  jointTravelTime = endTime - currTime
416  jointTravelDistance = endPos - currPos
417  jointIdealVelocity = float(jointTravelDistance) / jointTravelTime
418  _jointsControlData[jointName]["ideal_velocity"] = jointIdealVelocity
419  _jointsControlData[jointName]["bool_update_iv"] = False
420  else:
421  print("ERROR in compute ideal joint velocity: currTime > endTime")
422 
423  thisPosErrorDerivative = float(jointError - _jointsControlData[jointName]["prev_error"])/CONTROLLER_FREQUENCY
424 
425  #print("COMPUTED %s VELOCITY SETPOINT: %s (jointTravelTime: %s, jointTravelDistance: %s)" % (jointName, jointVelocityReachGoal, jointTravelTime, jointTravelDistance))
426 
427  thisReturnVal = _jointsControlData[jointName]["ideal_velocity"]
428  thisReturnVal = thisReturnVal*JOINT_VELOCITY_FACTOR
429  # SWITCH CONTROL MODE IF STATEMENT IS TRUE (FROM IDEAL VELOCITY TO PID POSITION ERROR)
430  if currTime > endTime: # jointError > JOINT_TRAJECTORY_ERROR_TOLERANCE or
431  if PRINT_DEBUG:
432  print("Switching control to PID for joint %s" % jointName)
433  _jointsControlData[jointName]["pos_error_integral"] += float(jointError)/CONTROLLER_FREQUENCY
434  _jointsControlData[jointName]["prev_time"] = currTime
435  _jointsControlData[jointName]["prev_pos"] = currPos
436  _jointsControlData[jointName]["prev_error"] = jointError
437  thisReturnVal = _jointsControlData[jointName]["param_p"]*jointError + _jointsControlData[jointName]["param_i"]*_jointsControlData[jointName]["pos_error_integral"] + _jointsControlData[jointName]["param_d"]*thisPosErrorDerivative
438 
439  thisReturnVal = checkOutputLimits(thisReturnVal)
440 
441  if PRINT_DEBUG:
442  if jointName == RIGHT_HIP_JOINT:
443  print("\t\t\t\t%0.5f\t\t\t\t%0.5f" % (jointError, thisReturnVal), end='\r')
444  elif jointName == RIGHT_KNEE_JOINT:
445  print("\t\t\t\t\t\t\t\t\t\t\t\t%0.5f\t\t\t\t%0.5f" % (jointError, thisReturnVal), end='\r')
446 
447  return thisReturnVal
