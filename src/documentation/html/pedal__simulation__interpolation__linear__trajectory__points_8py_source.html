403 def computeVelocitySetpoint(jointName, endPos, startTime, currTime, endTime):
404

413 
414  if _jointsControlData[jointName]["bool_update_iv"]:
415  if currTime < endTime:
416  jointTravelTime = endTime - currTime
417  jointTravelDistance = endPos - currPos
418  jointIdealVelocity = float(jointTravelDistance) / jointTravelTime
419  _jointsControlData[jointName]["ideal_velocity"] = jointIdealVelocity
420  _jointsControlData[jointName]["bool_update_iv"] = False
421  else:
422  print("ERROR in compute ideal joint velocity: currTime > endTime")
423 
424  thisPosErrorDerivative = float(jointError - _jointsControlData[jointName]["prev_error"])/CONTROLLER_FREQUENCY
425 
426  #print("COMPUTED %s VELOCITY SETPOINT: %s (jointTravelTime: %s, jointTravelDistance: %s)" % (jointName, jointVelocityReachGoal, jointTravelTime, jointTravelDistance))

429  thisReturnVal = thisReturnVal*JOINT_VELOCITY_FACTOR

431  if currTime > endTime: # jointError > JOINT_TRAJECTORY_ERROR_TOLERANCE or
432  if PRINT_DEBUG:
433  print("Switching control to PID for joint %s" % jointName)

437  _jointsControlData[jointName]["prev_error"] = jointError

439 
440  thisReturnVal = checkOutputLimits(thisReturnVal)
441 
442  if PRINT_DEBUG:
