371 def getCurrentAngle(current_point):
372  current_x = current_point[0] - PEDAL_CENTER_OFFSET_X
373  current_y = current_point[1] - PEDAL_CENTER_OFFSET_Y
374 
375  if current_x > 0 and current_y > 0:
376  return np.arctan(current_y / current_x)
377  elif current_x < 0 < current_y:
378  return np.arctan(current_y / current_x) + np.pi
379  elif current_x < 0 and current_y < 0:
380  return np.arctan(current_y / current_x) + np.pi
381  elif current_x > 0 > current_y:
382  return np.arctan(current_y / current_x) + 2 * np.pi
383 
384  elif current_x == 0 and current_y > 0:
385  return np.pi / 2
386  elif current_x == 0 and current_y < 0:
387  return np.pi * 3 / 2
388  elif current_x > 0 and current_y == 0:
389  return 0
390  elif current_x < 0 and current_y == 0:
391  return np.pi
