422 def computeVelocitySetpoint(jointName, next_joint_angle, current_joint_angle, startTime, currTime, endTime):
423  global _jointsControlData
424  global PRINT_DEBUG
425  global JOINT_TRAJECTORY_ERROR_TOLERANCE
426  global JOINT_VELOCITY_FACTOR
427 
428  current_point = getJointPosition(jointName)
429  joint_difference = next_joint_angle - current_joint_angle
430 
431  if _jointsControlData[ jointName ][ "bool_update_iv" ]:
432  if currTime < endTime:
433  jointTravelTime = endTime - currTime
434  jointTravelDistance = joint_difference
435  jointIdealVelocity = float(jointTravelDistance) / jointTravelTime
436  _jointsControlData[ jointName ][ "ideal_velocity" ] = jointIdealVelocity
437  _jointsControlData[ jointName ][ "bool_update_iv" ] = False
438  else:
439  print("ERROR in compute ideal joint velocity: currTime > endTime")
440 
441  thisPosErrorDerivative = float(
442  joint_difference - _jointsControlData[ jointName ][ "prev_error" ]) / CONTROLLER_FREQUENCY
443 
444  # print("COMPUTED %s VELOCITY SETPOINT: %s (jointTravelTime: %s, jointTravelDistance: %s)" % (jointName, jointVelocityReachGoal, jointTravelTime, jointTravelDistance))
445 
446  thisReturnVal = _jointsControlData[ jointName ][ "ideal_velocity" ]
447  thisReturnVal = thisReturnVal * JOINT_VELOCITY_FACTOR
448  # SWITCH CONTROL MODE IF STATEMENT IS TRUE (FROM IDEAL VELOCITY TO PID POSITION ERROR)
449  if currTime > endTime: # jointError > JOINT_TRAJECTORY_ERROR_TOLERANCE or
450  if PRINT_DEBUG:
451  print("Switching control to PID for joint %s" % jointName)
452  _jointsControlData[ jointName ][ "pos_error_integral" ] += float(joint_difference) / CONTROLLER_FREQUENCY
453  _jointsControlData[ jointName ][ "prev_time" ] = currTime
454  _jointsControlData[ jointName ][ "prev_pos" ] = current_point
455  _jointsControlData[ jointName ][ "prev_error" ] = joint_difference
456  thisReturnVal = _jointsControlData[ jointName ][ "param_p" ] * joint_difference + \
457  _jointsControlData[ jointName ][ "param_i" ] * _jointsControlData[ jointName ][
458  "pos_error_integral" ] + _jointsControlData[ jointName ][
459  "param_d" ] * thisPosErrorDerivative
460 
461  thisReturnVal = checkOutputLimits(thisReturnVal)
462 
463  if PRINT_DEBUG:
464  if jointName == RIGHT_HIP_JOINT:
465  print("\t\t\t\t%0.5f\t\t\t\t%0.5f" % (joint_difference, thisReturnVal), end='\r')
466  elif jointName == RIGHT_KNEE_JOINT:
467  print("\t\t\t\t\t\t\t\t\t\t\t\t%0.5f\t\t\t\t%0.5f" % (joint_difference, thisReturnVal), end='\r')
468 
469  return thisReturnVal
