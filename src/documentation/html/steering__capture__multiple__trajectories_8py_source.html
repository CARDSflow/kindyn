160 def computeSteeringAngles():
161  global _steeringAngles
162  global NUM_STEERING_ANGLES
163 
164  if NUM_STEERING_ANGLES % 2 == 0:
165  print("ERROR: NUM_STEERING_ANGLES must be odd number")
166  return
167 
168  numSymmetricSteeringAngles = (NUM_STEERING_ANGLES - 1) / 2
169 
170  for sizeIterator in range(NUM_STEERING_ANGLES):
171  _steeringAngles.append(0.0)
172 
173  _steeringAngles[int(NUM_STEERING_ANGLES / 2)] = 0.0 # Redundant
174  for steeringAngleIterator in range(1, numSymmetricSteeringAngles + 1, 1):
175  thisAngle = steeringAngleIterator * (MAX_TURNING_ANGLE / numSymmetricSteeringAngles)
176  _steeringAngles[numSymmetricSteeringAngles + steeringAngleIterator] = thisAngle
177  _steeringAngles[numSymmetricSteeringAngles - steeringAngleIterator] = (-1) * thisAngle

186  for steeringAngleIterator in range(len(_steeringAngles)):
