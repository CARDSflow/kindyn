159 def computeSteeringAngles():
160  global _steeringAngles

162 
163  if NUM_STEERING_ANGLES % 2 == 0:
164  print("ERROR: NUM_STEERING_ANGLES must be odd number")

172  _steeringAngles[int(NUM_STEERING_ANGLES / 2)] = 0.0 # Redundant
173  for steeringAngleIterator in range(1, numSymmetricSteeringAngles + 1, 1):
174  thisAngle = steeringAngleIterator * (MAX_TURNING_ANGLE / numSymmetricSteeringAngles)

176  _steeringAngles[numSymmetricSteeringAngles - steeringAngleIterator] = (-1) * thisAngle
