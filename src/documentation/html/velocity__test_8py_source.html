125 def evaluate_current_pedal_angle(current_point):
126  current_x = current_point[ 0 ] - PEDAL_CENTER_OFFSET_X
127  current_y = current_point[ 1 ] - PEDAL_CENTER_OFFSET_Y
128 
129  if current_x > 0 and current_y > 0:
130  return np.arctan(current_y / current_x)
131  elif current_x < 0 and current_y > 0:
132  return np.arctan(current_y / current_x) + np.pi
133  elif current_x < 0 and current_y < 0:
134  return np.arctan(current_y / current_x) + np.pi
135  elif current_x > 0 > current_y:
136  return np.arctan(current_y / current_x) + 2 * np.pi
137 
138  elif current_x == 0 and current_y > 0:
139  return np.pi / 2
140  elif current_x == 0 and current_y < 0:
141  return np.pi * 3 / 2
142  elif current_x > 0 and current_y == 0:
143  return 0
144  elif current_x < 0 and current_y == 0:
145  return np.pi
